defmodule ABZU.Sovereign do
  use Phoenix.LiveView

  # BOB AI inline completion
  def handle_event("complete", params, socket) do
    suggestion = BOB.Orchestrator.complete(params)
    {:noreply, assign(socket, :suggestion, suggestion)}
  end

  def seal_session(session_id) do
    WORM.Chain.append(%{
      type: :session_seal,
      id:   session_id,
      ts:   DateTime.utc_now()
    })
  end
end
